    # minimum - the smallest modifier is chosen, e.g. 0.1 and -0.1 = -0.1
    #     0 = no modifier, <0 = negative modifier, >0 = positive modifier
    # maximum - the largest modifier is chosen, e.g. 0.1 and -0.1 = 0.1
