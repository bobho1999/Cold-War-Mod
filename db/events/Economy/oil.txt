# oil 40 to 48

event = {
	id = 6548001
	persistent = yes
	country = USA #Only one country, so it will work in multiplayer
	random = no
	date = { day = 3 month = may year = 1003 }
	offset = 90
	deathdate = { day = 1 month = january year = 3000 }
	trigger = {
		#ai = no #will inform the player, but set things globally
		
		flag = oil_20
		NOT = { flag = oil_plus8 }
		NOT = { flag = oil_plus16 }
		NOT = { flag = oil_minus8 }
		NOT = { flag = oil_minus16 }
		OR = {
			AND = {
				flag = oil_tension1
				flag = oil_tension2
			}
			flag = oil_war_nearby
		}
		NOT = {
	 		OR = {
				flag = oil_occupied
				flag = oil_war
			} 
		}
	}
	name = "EVT_6548001_NAME"
	desc = "EVT_6548001_DESC"
	picture = "oil"
	style = 0
	action_a = {
		name = "OK"
		command = { type = setflag which = oil_plus8 }
		command = { type = clrflag which = oil_plus16 }
		command = { type = clrflag which = oil_minus16 }
		command = { type = clrflag which = oil_minus8 }
		command = { type = setflag which = oil_rises }
		command = { type = clrflag which = oil_tension1 }
		command = { type = clrflag which = oil_tension2 }
		command = { type = clrflag which = oil_war_nearby }

	}
}
